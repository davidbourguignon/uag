 10 /** @namespace uAg project */
 11 var uag = (function (parent, $, window, document) {
 12     'use strict';
 13     var uAgUtils = parent.utils = parent.utils || {};
 14 
 15     /**
 16      * @constructor
 17      * @exports uAgUtils.FileExplorer as uag.utils.fileExplorer
 18      * @description File explorer utility for enhancing controllers
 19      */
 20     uAgUtils.FileExplorer = function (viewBackBtn) {
 21         var view = {}; // jQuery objects
 22         var fileSystemRoot = null;
 23         var currentDir = null;
 24         var parentDir = null;
 25 
 26         if (viewBackBtn instanceof jQuery) {
 27             view.backBtn = viewBackBtn;
 28         } else {
 29             throw new TypeError('Error: wrong arguments for fileExplorer()');
 30         }
 31 
 32         this.initAfterCordova = function () {
 33             window.requestFileSystem(
 34                 LocalFileSystem.PERSISTENT,
 35                 0,
 36                 function (fileSystem) { // success
 37                     fileSystemRoot = fileSystem.root;
 38                 },
 39                 function (event) { // failure
 40                     throw new Error('Error: requestFileSystem() failed with code '
 41                                       + event.target.error.code);
 42                 }
 43             );
 44             view.backBtn.hide(); // view selection
 45         }
 46     }
 47 
 48     return parent;
 49 }(uag || {}, jQuery, this, this.document));
